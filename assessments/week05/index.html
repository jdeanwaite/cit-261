Local Storage API

The localStorage API allows you to persist data on a client's machine that will persist even through a browser restart. Session storage only allows you to store data until the browser is closed.

Arrays, Associative Arrays, and Objects

Arrays, associative arrays, and objects can all be stored on localStorage in a similar fashion. They should be stored by converting them to JSON using JSON.stringify() and then parsed back into javascript objects by using JSON.parse(). Modify the list below and click save to persist a new array.


function saveAssociativeArray() {
    var testAssociativeArray = [];
    testAssociativeArray["book"] = "Harry Potter";
    testAssociativeArray["city"] = "London";
    console.log("storing:", testAssociativeArray);
    localStorage.setItem("testAssociativeArray", JSON.stringify(testAssociativeArray));
}
        

function saveObject() {
    var testObject = {
        hello: "World",
        count: 1
    };
    console.log("storing:", testObject);
    localStorage.setItem("testObject", JSON.stringify(testObject));
}
